```python
# cases/_protocols.py
from __future__ import annotations
import dataclasses
from dataclasses import dataclass
from pathlib import Path
from typing import Any, Protocol, TypedDict, runtime_checkable

from core.common.secrets_mask import mask_text

_MAX_DEPTH = 50


class ScenarioResult(TypedDict):
    case_id: str
    summary_text: str
    output_files: list[Path]
    metrics: dict[str, Any]
    failures: list[dict[str, Any]]
    extras: dict[str, Any]


@runtime_checkable
class OCRBackend(Protocol):
    def extract(
        self,
        image_path: Path | str,
        *,
        model: str,
        schema: dict[str, Any] | None = None,
        prompt: str | None = None,  # R2-H1 — 외부 backend도 prompt 주입 가능
    ) -> dict[str, Any]: ...


@runtime_checkable
class AIBackend(Protocol):
    def chat(self, messages: list[dict[str, str]], *, model: str | None = None) -> str: ...


@runtime_checkable
class MessagingBackend(Protocol):
    def send_discord(self, content: str, *, level: str) -> None: ...
    def send_email(self, message: Any) -> None: ...


@dataclass(frozen=True)
class Backends:
    ocr: OCRBackend
    ai: AIBackend
    msg: MessagingBackend


def _mask_recursive(value: Any, depth: int = 0) -> Any:
    if depth > _MAX_DEPTH:
        return "<TRUNCATED:depth>"
    if isinstance(value, str):
        return mask_text(value)
    if isinstance(value, bytes):
        return f"<bytes:{len(value)}>"
    if isinstance(value, dict):
        return {k: _mask_recursive(v, depth + 1) for k, v in value.items()}
    if isinstance(value, (list, tuple)):
        return [_mask_recursive(v, depth + 1) for v in value]
    if isinstance(value, Path):
        return mask_text(str(value))  # R3-M3 — path도 mask
    if dataclasses.is_dataclass(value) and not isinstance(value, type):
        return _mask_recursive(dataclasses.asdict(value), depth + 1)  # R1-L3
    return value  # int/float/bool/None/Decimal 등은 그대로 (string으로 변환 안 함)


def serialize_result(r: ScenarioResult) -> dict[str, Any]:
    """디스크 직렬화용 (run.json). 모든 string/path/nested에 mask 자동 적용."""
    return {
        "case_id": r["case_id"],
        "summary_text": _mask_recursive(r["summary_text"]),
        "output_files": [_mask_recursive(str(p)) for p in r["output_files"]],
        "metrics": _mask_recursive(r["metrics"]),
        "failures": _mask_recursive(r["failures"]),
        "extras": _mask_recursive(r["extras"]),
    }


def as_display(r: ScenarioResult) -> dict[str, Any]:
    """Streamlit 화면 렌더용 — serialize_result와 동일한 sanitize 보장 (R1-C1).

    R1 audit C1 처방: 모든 화면 위젯은 이 함수 거친 dict만 받음.
    raw `result`를 st.dataframe / st.json에 직접 넘기는 것 금지.
    """
    return serialize_result(r)
```

```python
from __future__ import annotations
import hashlib
import json
from pathlib import Path
from typing import Any
from cases._protocols import OCRBackend, AIBackend, MessagingBackend


def _key(case_id: str, identity: str, args_repr: str) -> str:
    return hashlib.sha256(f"{case_id}|{identity}|{args_repr}".encode()).hexdigest()[:32]


class CachedOCRBackend:
    def __init__(self, real: OCRBackend, store: Path, *, case_id: str) -> None:
        self._real = real
        self._store = store
        self._case_id = case_id
        self._identity = real.cache_identity() if hasattr(real, "cache_identity") else "unknown"
        self._store.mkdir(parents=True, exist_ok=True)

    def cache_identity(self) -> str:
        return self._identity

    def extract(self, image_path: Path | str, *, model: str, schema: dict[str, Any] | None = None, prompt: str | None = None) -> dict[str, Any]:
        args_repr = json.dumps(
            {"img": str(image_path), "model": model, "schema": schema, "prompt": prompt},
            sort_keys=True, default=str,
        )
        key = _key(self._case_id, self._identity, args_repr)
        cache_file = self._store / f"{key}.json"
        if cache_file.exists():
            return json.loads(cache_file.read_text(encoding="utf-8"))
        result = self._real.extract(image_path, model=model, schema=schema, prompt=prompt)
        cache_file.write_text(json.dumps(result, ensure_ascii=False, default=str), encoding="utf-8")
        return result

    def __init__(self, real: AIBackend, store: Path, *, case_id: str) -> None:
        self._real = real
        self._store = store
        self._case_id = case_id

    def cache_identity(self) -> str:
        return self._identity

    def chat(self, messages: list[dict[str, str]], *, model: str | None = None) -> str:
        args_repr = json.dumps({"messages": messages, "model": model}, sort_keys=True, default=str)
        key = _key(self._case_id, self._identity, args_repr)
        cache_file = self._store / f"{key}.json"
        if cache_file.exists():
            return json.loads(cache_file.read_text(encoding="utf-8"))["response"]
        result = self._real.chat(messages, model=model)
        cache_file.write_text(json.dumps({"response": result}, ensure_ascii=False), encoding="utf-8")
        return result


class CachedMessagingBackend:
    """Messaging은 부수효과 (외부 발송) — cache는 idempotency 추적만. 같은 args 두 번이면 두 번째는 skip."""

        return self._identity

    def send_discord(self, content: str, *, level: str) -> None:

        if marker.exists():
            return  # idempotent skip
        self._real.send_discord(content, level=level)
        marker.write_text("", encoding="utf-8")

    def send_email(self, message: Any) -> None:
```

```python
from __future__ import annotations
import os
```

```python
_SAFE_VAR: ContextVar[bool] = ContextVar("safe_mode_v2", default=os.getenv("DEMO_SAFE") == "1")
```

```python
# core/progress.py
from __future__ import annotations
from collections.abc import Callable
from dataclasses import dataclass


@dataclass(frozen=True, slots=True)
class ProgressEvent:
    processed: int
    total: int
    label: str
    case_id: str


def rich_progress_adapter(description: str) -> Callable[[ProgressEvent], None]:
    from rich.progress import Progress, BarColumn, TextColumn, TimeRemainingColumn
    progress = Progress(
        TextColumn("[progress.description]{task.description}"),
        BarColumn(),
        TextColumn("[progress.percentage]{task.percentage:>3.0f}%"),
        TimeRemainingColumn(),
    )
    progress.start()
    task_id: int | None = None

    def cb(evt: ProgressEvent) -> None:
        nonlocal task_id
        if task_id is None:
            task_id = progress.add_task(description, total=evt.total)
        progress.update(task_id, completed=evt.processed, description=f"{description}: {evt.label}")
        if evt.processed >= evt.total:
            progress.stop()

    return cb
```

```python
# scripts/migrate_imports.py
"""core.X → flowcoder_office_tools.X 자동 변환. ImportFrom + Import + string literal scope."""
import sys
from pathlib import Path
import libcst as cst
from libcst.codemod import CodemodContext, VisitorBasedCodemodCommand


def _replace_core(text: str) -> str:
    """`core.X` → `flowcoder_office_tools.X`, `core` 단독 → `flowcoder_office_tools`."""
    if text == "core" or text.startswith("core."):
        return "flowcoder_office_tools" + text[len("core"):]
    return text


class CoreToFotImports(VisitorBasedCodemodCommand):
    def leave_ImportFrom(self, original_node, updated_node):
        if updated_node.module is None:
            return updated_node
        text = cst.Module([]).code_for_node(updated_node.module)
        new_text = _replace_core(text)
        if new_text != text:
            return updated_node.with_changes(module=cst.parse_expression(new_text))
        return updated_node

    def leave_Import(self, original_node, updated_node):
        # `import core.X.Y as Z` 패턴
        new_names = []
        changed = False
        for alias in updated_node.names:
            text = cst.Module([]).code_for_node(alias.name)
            new_text = _replace_core(text)
            if new_text != text:
                new_names.append(alias.with_changes(name=cst.parse_expression(new_text)))
                changed = True
            else:
                new_names.append(alias)
        if changed:
            return updated_node.with_changes(names=new_names)
        return updated_node

    def leave_SimpleString(self, original_node, updated_node):
        """string literal "core.X" 변환 — monkeypatch / patch / import_module 인자."""
        raw = updated_node.value
        # quote 형식 보존
        for q in ("\"", "'"):
            if raw.startswith(q) and raw.endswith(q):
                inner = raw[1:-1]
                if inner == "core" or inner.startswith("core."):
                    new_inner = _replace_core(inner)
                    return updated_node.with_changes(value=f"{q}{new_inner}{q}")
        return updated_node


def main():
    targets = [p for p in Path(".").rglob("*.py") if "/.venv/" not in str(p) and "/__pycache__/" not in str(p)]
    for p in targets:
        src = p.read_text(encoding="utf-8")
        if "core" not in src:
            continue
        try:
            module = cst.parse_module(src)
        except Exception as e:
            print(f"[skip] {p}: {e}")
            continue
        new_module = CoreToFotImports(CodemodContext()).transform_module(module)
        if new_module.code != src:
            p.write_text(new_module.code, encoding="utf-8")
            print(f"[migrated] {p}")


if __name__ == "__main__":
    main()
```

```python
# tests/test_public_api_surface.py
import inspect
import json
import dataclasses
from pathlib import Path
from importlib import import_module
from typing import get_type_hints

PUBLIC_MODULES = [
    "flowcoder_office_tools.excel",
    "flowcoder_office_tools.messaging",
    "flowcoder_office_tools.docgen",
    "flowcoder_office_tools.ocr",
    "flowcoder_office_tools.ai",
    "flowcoder_office_tools.common",
    "flowcoder_office_tools.backends",
    "flowcoder_office_tools.protocols",
]

SNAPSHOT_PATH = Path(__file__).parent / "snapshots" / "public_api.json"


def _surface_repr(obj) -> str:
    """callable이면 signature, dataclass/TypedDict면 fields, 그 외 type 이름."""
    try:
        return str(inspect.signature(obj))
    except (TypeError, ValueError):
        pass
    if dataclasses.is_dataclass(obj):
        return f"dataclass({sorted(f.name for f in dataclasses.fields(obj))})"
    try:
        hints = get_type_hints(obj)
        return f"TypedDict({sorted(hints.keys())})"
    except Exception:
        pass
    return f"<{type(obj).__name__}>"


def _capture_surface() -> dict[str, dict[str, str]]:
    surface: dict[str, dict[str, str]] = {}
    for mod_name in PUBLIC_MODULES:
        mod = import_module(mod_name)
        all_list = getattr(mod, "__all__", None)
        assert all_list is not None, f"{mod_name} missing __all__"
        for name in all_list:
            assert not name.startswith("_"), f"{mod_name}.{name} underscore in __all__"
        surface[mod_name] = {name: _surface_repr(getattr(mod, name)) for name in all_list}
    return surface


def test_public_api_matches_snapshot():
    actual = _capture_surface()
    expected = json.loads(SNAPSHOT_PATH.read_text(encoding="utf-8"))
    if actual != expected:
        raise AssertionError(
            f"Public API surface changed. PR description에 'BREAKING CHANGE: ...' 명시 후 "
            f"snapshot 갱신 commit. Diff:\n{json.dumps({'expected': expected, 'actual': actual}, indent=2)}"
        )


def test_no_underscore_in_all():
    for mod_name in PUBLIC_MODULES:
        mod = import_module(mod_name)
        for name in getattr(mod, "__all__", []):
            assert not name.startswith("_"), f"{mod_name}.{name} 차단"
```

```python
# web/_runs.py
from __future__ import annotations
import secrets
import shutil
import time
from pathlib import Path

ALLOWED_EXTS = {".xlsx", ".csv", ".pdf", ".docx", ".hwpx", ".png", ".jpg", ".jpeg", ".txt", ".md", ".json"}
_ACTIVE_RUNS: set[Path] = set()


def create_run_dir(runs_root: Path) -> Path:
    runs_root.mkdir(parents=True, exist_ok=True)
    run_id = secrets.token_urlsafe(16)
    run_dir = runs_root / run_id
    (run_dir / "input").mkdir(parents=True)
    (run_dir / "output").mkdir(parents=True)
    return run_dir


def validate_upload_path(runs_root: Path, target: Path) -> Path:
    resolved = target.resolve()
    root_resolved = runs_root.resolve()
    if not resolved.is_relative_to(root_resolved):
        raise ValueError(f"path traversal — {target} resolved outside {runs_root}")
    if resolved.suffix.lower() not in ALLOWED_EXTS:
        raise ValueError(f"extension {resolved.suffix} not allowed")
    return resolved


def stream_save(uf, target: Path, *, per_file_mb: int = 50, chunk_size: int = 1024 * 1024) -> int:
    """R1-H1+H2: streaming write + 사전/실시간 cap 검증."""
    cap = per_file_mb * 1024 * 1024
    size = 0
    with target.open("wb") as fh:
        while True:
            chunk = uf.read(chunk_size)
            if not chunk:
                break
            size += len(chunk)
            if size > cap:
                fh.close()
                target.unlink(missing_ok=True)
                raise ValueError(f"file {target.name} exceeds {per_file_mb}MB cap")
            fh.write(chunk)
    return size


def mark_active(run_dir: Path) -> None:
    """R1-H4: lock file로 active mark."""
    _ACTIVE_RUNS.add(run_dir.resolve())
    (run_dir / ".lock").write_text(str(time.time()), encoding="utf-8")


def mark_done(run_dir: Path) -> None:
    _ACTIVE_RUNS.discard(run_dir.resolve())
    lock = run_dir / ".lock"
    if lock.exists():
        lock.unlink()


def cleanup_expired_runs(runs_root: Path, *, ttl_hours: int = 24) -> int:
    if not runs_root.exists():
        return 0
    cutoff = time.time() - ttl_hours * 3600
    removed = 0
    for run_dir in runs_root.iterdir():
        if not run_dir.is_dir():
            continue
        if run_dir.resolve() in _ACTIVE_RUNS:
            continue  # active skip
        if (run_dir / ".lock").exists():
            continue  # 다른 process active
        if run_dir.stat().st_mtime < cutoff:
            shutil.rmtree(run_dir)
            removed += 1
    return removed
```

```python
    run_dir = create_run_dir(RUNS_ROOT)
    in_dir = run_dir / "input"
    out_dir = run_dir / "output"

        # 업로드 파일 streaming write + 검증

        backs = safe_backends() if safe_mode else default_backends()

                input_dir=in_dir if list(in_dir.iterdir()) else None,
                output_dir=out_dir,
                backends=backs,
                config=config or None,
            )

        run_json.write_text(
            json.dumps(serialize_result(result), ensure_ascii=False, indent=2),
```
