```python
# cases/_protocols.py
from __future__ import annotations
from dataclasses import dataclass
from pathlib import Path
from typing import Any, Protocol, TypedDict, runtime_checkable

from core.common.secrets_mask import mask_text


class ScenarioResult(TypedDict):
    case_id: str
    summary_text: str
    output_files: list[Path]
    metrics: dict[str, Any]
    failures: list[dict[str, Any]]
    extras: dict[str, Any]


@runtime_checkable
class OCRBackend(Protocol):
    def extract(self, image_path: Path, *, model: str, schema: dict[str, Any] | None) -> dict[str, Any]: ...


@runtime_checkable
class AIBackend(Protocol):
    def chat(self, messages: list[dict[str, str]], *, model: str | None = None) -> str: ...


@runtime_checkable
class MessagingBackend(Protocol):
    def send_discord(self, content: str, *, level: str) -> None: ...
    def send_email(self, message: Any) -> None: ...


@dataclass(frozen=True)
class Backends:
    ocr: OCRBackend
    ai: AIBackend
    msg: MessagingBackend


def _mask_recursive(value: Any) -> Any:
    if isinstance(value, str):
        return mask_text(value)
    if isinstance(value, dict):
        return {k: _mask_recursive(v) for k, v in value.items()}
    if isinstance(value, list):
        return [_mask_recursive(v) for v in value]
    if isinstance(value, Path):
        return str(value)
    return value


def serialize_result(r: ScenarioResult) -> dict[str, Any]:
    return {
        "case_id": r["case_id"],
        "summary_text": _mask_recursive(r["summary_text"]),
        "output_files": [str(p) for p in r["output_files"]],
        "metrics": _mask_recursive(r["metrics"]),
        "failures": _mask_recursive(r["failures"]),
        "extras": _mask_recursive(r["extras"]),
    }
```

```python
import hashlib
import json
from pathlib import Path
from typing import Any
from cases._protocols import OCRBackend


def _fingerprint(backend: Any) -> str:
    """backend 인스턴스의 식별 fingerprint (api_key SHA256 prefix 등)."""
    raw = repr(sorted(getattr(backend, "__dict__", {}).items()))
    return hashlib.sha256(raw.encode()).hexdigest()[:8]


def _key(case_id: str, fp: str, args_repr: str) -> str:
    return hashlib.sha256(f"{case_id}|{fp}|{args_repr}".encode()).hexdigest()[:32]


class CachedOCRBackend:
    def __init__(self, real: OCRBackend, store: Path, *, case_id: str) -> None:
        self._real = real
        self._store = store
        self._case_id = case_id
        self._fp = _fingerprint(real)
        self._store.mkdir(parents=True, exist_ok=True)

    def extract(self, image_path: Path, *, model: str, schema: dict[str, Any] | None) -> dict[str, Any]:
        args_repr = json.dumps({"img": str(image_path), "model": model, "schema": schema}, sort_keys=True, default=str)
        key = _key(self._case_id, self._fp, args_repr)
        cache_file = self._store / f"{key}.json"
        if cache_file.exists():
            return json.loads(cache_file.read_text(encoding="utf-8"))
        result = self._real.extract(image_path, model=model, schema=schema)
        cache_file.write_text(json.dumps(result, ensure_ascii=False, default=str), encoding="utf-8")
        return result
```

```python
# core/common/safe_mode_v2.py
from __future__ import annotations
import os
from contextlib import contextmanager
from contextvars import ContextVar
from typing import Iterator

_SAFE_VAR: ContextVar[bool] = ContextVar("safe_mode_v2", default=os.getenv("DEMO_SAFE") == "1")


def is_safe() -> bool:
    return _SAFE_VAR.get()


@contextmanager
def safe_mode_scope(enabled: bool = True) -> Iterator[None]:
    token = _SAFE_VAR.set(enabled)
    try:
        yield
    finally:
        _SAFE_VAR.reset(token)


def force_safe() -> None:
    """Backend 폴백 시 호출. ContextVar로만 set, os.environ mutation 없음."""
    _SAFE_VAR.set(True)
```

```python
# core/progress.py
from __future__ import annotations
from collections.abc import Callable
from dataclasses import dataclass
from typing import TYPE_CHECKING

if TYPE_CHECKING:
    from rich.progress import Progress


@dataclass(frozen=True, slots=True)
class ProgressEvent:
    processed: int
    total: int
    label: str
    case_id: str


def rich_progress_adapter(description: str) -> Callable[[ProgressEvent], None]:
    """CLI: rich.progress.Progress wrapper. lazily creates progress bar."""
    from rich.progress import Progress, BarColumn, TextColumn, TimeRemainingColumn

    progress = Progress(
        TextColumn("[progress.description]{task.description}"),
        BarColumn(),
        TextColumn("[progress.percentage]{task.percentage:>3.0f}%"),
        TimeRemainingColumn(),
    )
    progress.start()
    task_id: int | None = None

    def cb(evt: ProgressEvent) -> None:
        nonlocal task_id
        if task_id is None:
            task_id = progress.add_task(description, total=evt.total)
        progress.update(task_id, completed=evt.processed, description=f"{description}: {evt.label}")
        if evt.processed >= evt.total:
            progress.stop()

    return cb
```

```python
# scripts/migrate_imports.py
"""core.X → flowcoder_office_tools.X 자동 변환."""
import sys
from pathlib import Path
import libcst as cst
from libcst.codemod import CodemodContext, VisitorBasedCodemodCommand


class CoreToFotImports(VisitorBasedCodemodCommand):
    def leave_ImportFrom(self, original_node, updated_node):
        module = updated_node.module
        if module is None:
            return updated_node
        # ImportFrom 모듈 이름 추출 (Attribute / Name)
        text = cst.Module([]).code_for_node(module)
        if text == "core" or text.startswith("core."):
            new_text = "flowcoder_office_tools" + text[len("core"):]
            new_module = cst.parse_expression(new_text)
            return updated_node.with_changes(module=new_module)
        return updated_node

    def leave_Import(self, original_node, updated_node):
        # `import core.X` 패턴은 본 프로젝트에 거의 없음 — skip 가능
        return updated_node


def main():
    targets = list(Path(".").rglob("*.py"))
    targets = [p for p in targets if "/.venv/" not in str(p) and "/__pycache__/" not in str(p)]
    for p in targets:
        src = p.read_text(encoding="utf-8")
        if "core" not in src:
            continue
        try:
            module = cst.parse_module(src)
        except Exception:
            continue
        context = CodemodContext()
        new_module = CoreToFotImports(context).transform_module(module)
        if new_module.code != src:
            p.write_text(new_module.code, encoding="utf-8")
            print(f"[migrated] {p}")


if __name__ == "__main__":
    main()
```

```python
        }

            "removed": {k: v for k, v in expected.items() if k not in actual},
            "changed": {
```

```python
# web/_runs.py
from __future__ import annotations
import secrets
from pathlib import Path

ALLOWED_EXTS = {".xlsx", ".csv", ".pdf", ".docx", ".hwpx", ".png", ".jpg", ".jpeg", ".txt", ".md"}


def create_run_dir(runs_root: Path) -> Path:
    runs_root.mkdir(parents=True, exist_ok=True)
    run_id = secrets.token_urlsafe(16)
    run_dir = runs_root / run_id
    (run_dir / "input").mkdir(parents=True)
    (run_dir / "output").mkdir(parents=True)
    return run_dir


def validate_upload_path(runs_root: Path, target: Path) -> Path:
    resolved = target.resolve()
    root_resolved = runs_root.resolve()
    if not resolved.is_relative_to(root_resolved):
        raise ValueError(f"path traversal — {target} resolved outside runs_root")
    if resolved.suffix.lower() not in ALLOWED_EXTS:
        raise ValueError(f"extension {resolved.suffix} not allowed")
    return resolved


def enforce_size_cap(file_path: Path, *, per_file_mb: int = 50) -> None:
    size = file_path.stat().st_size
    if size > per_file_mb * 1024 * 1024:
        raise ValueError(f"file {file_path.name} exceeds {per_file_mb}MB cap")
```

```python
import importlib
import json
from pathlib import Path
from flowcoder_office_tools.protocols import serialize_result
from flowcoder_office_tools.backends import safe_backends, default_backends
from flowcoder_office_tools.common.safe_mode_v2 import safe_mode_scope
from web._runs import create_run_dir, validate_upload_path, enforce_size_cap

RUNS_ROOT = Path("runs")


def execute_case(case_id: str, uploaded_files: list, safe_mode: bool) -> dict:
    run_dir = create_run_dir(RUNS_ROOT)
    in_dir = run_dir / "input"
    out_dir = run_dir / "output"

    # 업로드 저장 + 검증
    total_bytes = 0
    for uf in uploaded_files:
        target = in_dir / uf.name
        validate_upload_path(RUNS_ROOT, target)
        total_bytes += len(uf.getvalue())
        if total_bytes > 200 * 1024 * 1024:
            raise ValueError("total upload exceeds 200MB cap")
        target.write_bytes(uf.getvalue())
        enforce_size_cap(target, per_file_mb=50)

    # scenario import + 실행
    scenario_mod = importlib.import_module(f"cases.{case_id}.scenario")
    backs = safe_backends() if safe_mode else default_backends()
    with safe_mode_scope(safe_mode):
        result = scenario_mod.run(input_dir=in_dir, output_dir=out_dir, backends=backs)

    # ScenarioResult → run.json
    run_json = run_dir / "run.json"
    run_json.write_text(json.dumps(serialize_result(result), ensure_ascii=False, indent=2), encoding="utf-8")
    return {"run_id": run_dir.name, "result": result, "run_dir": run_dir}
```

```python
from collections.abc import Callable
import streamlit as st
from flowcoder_office_tools.progress import ProgressEvent


def streamlit_progress_adapter(placeholder: "st.delta_generator.DeltaGenerator") -> Callable[[ProgressEvent], None]:
    bar = placeholder.progress(0, text="시작 대기 중")

    def cb(evt: ProgressEvent) -> None:
        pct = int(evt.processed * 100 / max(evt.total, 1))
        bar.progress(pct, text=f"{evt.case_id}: {evt.processed}/{evt.total} — {evt.label}")

    return cb
```

```python
# web/_runs.py — 추가
import time


def cleanup_expired_runs(runs_root: Path, *, ttl_hours: int = 24) -> int:
    if not runs_root.exists():
        return 0
    cutoff = time.time() - ttl_hours * 3600
    removed = 0
    for run_dir in runs_root.iterdir():
        if not run_dir.is_dir():
            continue
        if run_dir.stat().st_mtime < cutoff:
            import shutil
            shutil.rmtree(run_dir)
            removed += 1
    return removed
```
